示例：

    //#pragma pack(4) 
    #include <stdio.h>
    #define offsetof(TYPE, MEMBER) ((size_t ) &((TYPE *)0)->MEMBER)
    typedef struct s
    {
        /* data */
        union {
            int a;//4字节
            char str[10];//10字节
        };
        struct s* next;
    }S;

    int main(){
        printf("%ld",offsetof(S, next));
        return 0;    
    }

结果为16，因为电脑是64位的，所以以8字节对齐，联合体字节大小以最大的成员变量为准，故联合体的字节大小是10个字节
