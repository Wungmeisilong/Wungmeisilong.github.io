推送网站步骤

推送网站步骤

    #在项目更目录下运行下面这句命令
    hugo --theme=hugo-theme-relearn-main --baseURL="https://wungmeisilong.github.io/" --buildDrafts

    #进入public文件夹
    cd  public/

    #初始化文件夹，这个命令不一定要需要，
    #有时候不初始化也可以推送，之前执行过久不建议在执行
    git init

    #加入要推送的文件
    git add .

    #添加更改备注文字
    git commit -m "备注的文字"

    #推送到远程仓库
    git push origin main
